def send_prompt_and_get_response(driver, prompt_text):
    # Find and fill the prompt input
    prompt_input = driver.find_element(By.CSS_SELECTOR, "your-prompt-input-selector")
    prompt_input.clear()
    prompt_input.send_keys(prompt_text)
    
    # Find and click submit button
    submit_button = driver.find_element(By.CSS_SELECTOR, "your-submit-button-selector")
    submit_button.click()
    
    # Wait for response to appear
    try:
        # Wait for the response container to appear
        response_element = WebDriverWait(driver, 60).until(
            EC.presence_of_element_located((By.CSS_SELECTOR, "response-container-selector"))
        )
        
        # Wait for response to complete (look for specific indicators)
        WebDriverWait(driver, 120).until(
            EC.text_to_be_present_in_element((By.CSS_SELECTOR, "thinking-indicator"), "specific-text-when-done")
        )
        
        # Get the response text
        response_text = response_element.text
        return response_text
        
    except Exception as e:
        print(f"Error getting response: {e}")
        return None

async def wait_for_marked_response(page, timeout=60):
    """Wait for a response starting with ..s.."""
    start_time = time.time()
    last_response = ""
    
    while time.time() - start_time < timeout:
        # Get all message elements (adjust selector based on actual page structure)
